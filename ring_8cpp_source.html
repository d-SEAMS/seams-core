101 bool ring::findTripletInRing(std::vector<int> ring, std::vector<int> triplet) {
102  //
103  int ringSize = ring.size(); // should be 6
104  std::vector<int> ringTriplet; // triplet from the ring to be searched
105  int kIndex; // Used for making the triplet
106 
107  // Loop through every possible triplet in the ring to be searched
108  for (int i = 0; i < ringSize; i++) {
109  ringTriplet.clear(); // init
110  // Get the first element of the ring triplet
111  ringTriplet.push_back(ring[i]);
112  //
113  // Get the next two elements
114  for (int k = 1; k < 3; k++) {
115  kIndex = i + k;
116  // Wrap-around
117  if (kIndex >= ringSize) {
118  kIndex -= ringSize;
119  } // end of wrap-around
120  ringTriplet.push_back(ring[kIndex]);
121  } // next two elements of ringTriplet
122  //
123  // Obtained ringTriplet!
124  // Check equality
125  if (triplet == ringTriplet) {
126  return true;
127  } // triplet matches!
128  //

130  std::reverse(ringTriplet.begin(), ringTriplet.end());
131 
132  // Check for equality
133  if (triplet == ringTriplet) {
134  return true;
135  } // reversed triplet matches
136  } // first element of ringTriplet
137 
138  return false;
