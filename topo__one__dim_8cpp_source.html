778 int ring::rmAxialTranslations(
779  molSys::PointCloud<molSys::Point<double>, double> *yCloud, int *atomID,
780  int firstFrame, int currentFrame) {
781  //
782  int atomIndex; // Index of the atom to be shifted first
783  double shiftDistance; // Value by which all atoms will be shifted
784  double distFrmBoundary; // Distance from either the upper or lower boundary
785  // along the axial dimension
786  // Get the axial dimension
787  int axialDim = std::max_element(yCloud->box.begin(), yCloud->box.end()) -
788  yCloud->box.begin();
789  // Check: (default z)
790  if (axialDim < 0 || axialDim > 2) {
791  axialDim = 2;
792  } // end of check to set the axial dimension
793  // Lower and higher limits of the box in the axial dimension
794  double boxLowAxial = yCloud->boxLow[axialDim];
795  double boxHighAxial = boxLowAxial + yCloud->box[axialDim];
796  //
797  // -----------------------------------
798  // Update the atomID of the 'first' atom in yCloud if the current frame is the
799  // first frame.
800  // Get the atom index of the first atom to be shifted down or up
801  if (currentFrame == firstFrame) {
802  *atomID = yCloud->pts[0].atomID;
803  atomIndex = 0;
804  } // end of update of the atom ID to be shifted for the first frame
805  else {
806  //
807  int iatomID = *atomID; // Atom ID of the first particle to be moved down
808  // Find the index given the atom ID
809  auto it = yCloud->idIndexMap.find(iatomID);
810 
811  if (it != yCloud->idIndexMap.end()) {
812  atomIndex = it->second;
813  } // found jatom
814  else {
815  std::cerr << "Lost atoms.\n";
816  return 1;
817  } // error handling
818  //
819  } // Update of atomIndex for all other frames
820  // -----------------------------------
821  // Calculate the distance by which the atoms must be shifted (negative value)
822  if (axialDim == 0) {
823  shiftDistance = boxLowAxial - yCloud->pts[atomIndex].x;
824  } // x coordinate
825  else if (axialDim == 1) {
826  shiftDistance = boxLowAxial - yCloud->pts[atomIndex].y;
827  } // y coordinate
828  else {
829  shiftDistance = boxLowAxial - yCloud->pts[atomIndex].z;
830  } // z coordinate
831  // -----------------------------------
832  // Shift all the particles
833  for (int iatom = 0; iatom < yCloud->nop; iatom++) {
834  // Shift the particles along the axial dimension only
835  if (axialDim == 0) {
836  yCloud->pts[iatom].x += shiftDistance;
837  // Wrap-around
838  if (yCloud->pts[iatom].x < boxLowAxial) {
839  distFrmBoundary = boxLowAxial - yCloud->pts[iatom].x; // positive value
840  yCloud->pts[iatom].x = boxHighAxial - distFrmBoundary;
841  } // end of wrap-around
842  } // x coordinate
843  else if (axialDim == 1) {
844  yCloud->pts[iatom].y += shiftDistance;
845  // Wrap-around
846  if (yCloud->pts[iatom].y < boxLowAxial) {
847  distFrmBoundary = boxLowAxial - yCloud->pts[iatom].y; // positive value
848  yCloud->pts[iatom].y = boxHighAxial - distFrmBoundary;
849  } // end of wrap-around
850  } // y coordinate
851  else {
852  yCloud->pts[iatom].z += shiftDistance;
853  // Wrap-around
854  if (yCloud->pts[iatom].z < boxLowAxial) {
855  distFrmBoundary = boxLowAxial - yCloud->pts[iatom].z; // positive value
856  yCloud->pts[iatom].z = boxHighAxial - distFrmBoundary;
857  } // end of wrap-around
858  } // z coordinate
859  } // end of shifting all paritcles downward
860  // -----------------------------------
861  return 0;
862 } // end of function
