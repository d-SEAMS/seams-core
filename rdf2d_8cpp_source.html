149  std::vector<int> histogram, double binwidth, int nbin,
150  std::vector<double> volumeLengths, int nIter) {
151  //
152  auto height = *std::min_element(
153  volumeLengths.begin(),
154  volumeLengths.end()); // The height or the smallest dimension
155  // double planeArea = rdf2::getPlaneArea(
156  // volumeLengths); // The area of the quasi-two-dimensional water
157  double r; // Distance for the current bin
158  double factor; // Factor for accounting for the two-dimensional slab
159  double binVolume; // Volume of the current bin
160  double volumeDensity =
161  nopA / (volumeLengths[0] * volumeLengths[1] * volumeLengths[2]);
162 
163  // Loop through every bin
164  for (int ibin = 0; ibin < nbin; ibin++) {
165  //
166  r = binwidth * (ibin + 0.5);
167  // ----------------------
168  // Factor calculation
169  factor = 1;
170  if (r > height) {
171  // FACTOR = HEIGHT/(2*R)
172  factor = height / (2 * r);
173  } // r>height
174  else {
175  // FACTOR = 1 - R/(2*HEIGHT)
176  factor = 1 - r / (2 * height);
177  } // r<=height
178  // ----------------------
179  // binVolume = 4.0*PI*(DELTA_R**3)*((I_BIN+1)**3-I_BIN**3)/3.0
180  binVolume = 4 * gen::pi * pow(binwidth, 3.0) *
181  (pow(ibin + 1, 3.0) - pow(ibin, 3.0)) / 3.0;
182  // Update the RDF
183  (*rdfValues)[ibin] +=
184  histogram[ibin] / (nIter * binVolume * nopA * volumeDensity * factor);
185  } // end of loop through every bin
