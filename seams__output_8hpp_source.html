52 inline int makePath(const std::string &path) {
53 #if defined(_WIN32)
54  int ret = _mkdir(path.c_str());
55 #else
56  mode_t mode = 0755;
57  int ret = mkdir(path.c_str(), mode);
58 #endif
59  if (ret == 0) return 0;
60 
61  switch (errno) {
62  case ENOENT:
63  // parent didn't exist, try to create it
64  {
65  int pos = path.find_last_of('/');
66  if (pos == std::string::npos)
67 #if defined(_WIN32)
68  pos = path.find_last_of('\\');
69  if (pos == std::string::npos)
70 #endif
71  return 1;
72  if (!makePath(path.substr(0, pos))) return 1;
73  }
74 // now, try to create again
75 #if defined(_WIN32)
76  return 0 == _mkdir(path.c_str());
77 #else
78  return 0 == mkdir(path.c_str(), mode);
79 #endif
80 
81  case EEXIST:
82  // done!
83  if (isDirExist(path)) {
84  return 0;
85  } else {
86  return 1;
87  }
88 
89  default:
90  return 1;
91  }
